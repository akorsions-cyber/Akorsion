const STORE_KEY = 'portfolio_data';

function loadData() {
  try {
    return JSON.parse(localStorage.getItem(STORE_KEY)) || {};
  } catch(e) { return {}; }
}

function saveData(obj) {
  const current = loadData();
  const merged = Object.assign({}, current, obj);
  localStorage.setItem(STORE_KEY, JSON.stringify(merged));
}

// Add project
function addProject() {
  const d = loadData();
  const projects = d.projects || [];
  const title = getVal('a-proj-title');
  if(!title) return;
  projects.push({
    id: Date.now(),
    title,
    cat: getVal('a-proj-cat'),
    desc: getVal('a-proj-desc'),
    tech: getVal('a-proj-tech')
  });
  saveData({ projects });
  renderAdminProjects(projects);
  addProjectToGrid(projects[projects.length-1]);
  document.getElementById('a-proj-title').value = '';
  document.getElementById('a-proj-desc').value = '';
  document.getElementById('a-proj-tech').value = '';
  showToast();
}

function renderAdminProjects(projects) {
  const el = document.getElementById('admin-proj-list');
  if(!projects.length) { el.innerHTML = '<p style="color:var(--text3);font-size:0.85rem">No custom projects yet.</p>'; return; }
  el.innerHTML = projects.map(p => `<div class="admin-proj-item"><span>${p.title} <em style="color:var(--text3)">(${p.cat})</em></span><button onclick="deleteProject(${p.id})">✕ Remove</button></div>`).join('');
}

function deleteProject(id) {
  const d = loadData();
  const projects = (d.projects || []).filter(p => p.id !== id);
  saveData({ projects });
  renderAdminProjects(projects);
  const el = document.querySelector(`[data-proj-id="${id}"]`);
  if(el) el.remove();
  showToast();
}

function initTyping() {
  const d = loadData();
  const titles = d.titles && d.titles.length
    ? d.titles
    : ['Customer Service Expert', 'Team Leader', 'Problem Solver', 'Communication Pro'];
  const el = document.getElementById('hero-typed');
  let ti = 0, ci = 0, del = false;

  function type() {
    const word = titles[ti];
    if(!del) {
      el.innerHTML = word.substring(0, ci) + '<span class="typing-cursor">|</span>';
      ci++;
      if(ci > word.length) { del = true; setTimeout(type, 1800); return; }
    } else {
      el.innerHTML = word.substring(0, ci) + '<span class="typing-cursor">|</span>';
      ci--;
      if(ci < 0) { del = false; ti = (ti+1) % titles.length; ci = 0; }
    }
    setTimeout(type, del ? 60 : 100);
  }
  type();
}

  document.getElementById('live-clock').textContent =
    `${days[now.getDay()]}, ${months[now.getMonth()]} ${now.getDate()} — ${h}:${m}:${s}`;
}

//  INTERSECTION OBSERVER — fade-up & skill bars
// ============================================================
const io = new IntersectionObserver((entries) => {
  entries.forEach(e => {
    if(e.isIntersecting) {
      e.target.classList.add('visible');
      // Skill bars inside
      e.target.querySelectorAll('.skill-fill').forEach(bar => {
        bar.style.width = bar.getAttribute('data-width') + '%';
      });
      io.unobserve(e.target);
    }
  });
}, { threshold: 0.15 });

document.querySelectorAll('.fade-up').forEach(el => io.observe(el));

// Trigger skill bars when skills section enters view
const skillsIO = new IntersectionObserver(([entry]) => {
  if(entry.isIntersecting) {
    document.querySelectorAll('.skill-fill').forEach(bar => {
      bar.style.width = bar.getAttribute('data-width') + '%';
    });
    skillsIO.unobserve(entry.target);
  }
}, { threshold: 0.2 });
const skillsSec = document.getElementById('skills');
if(skillsSec) skillsIO.observe(skillsSec);

// ============================================================
//  COUNTER ANIMATION
// ============================================================
const counterIO = new IntersectionObserver(([entry]) => {
  if(entry.isIntersecting) {
    document.querySelectorAll('.counter').forEach(el => {
      const target = +el.getAttribute('data-target');
      let cur = 0;
      const step = target / 60;
      const t = setInterval(() => {
        cur = Math.min(cur + step, target);
        el.textContent = Math.floor(cur) + (el.nextElementSibling?.textContent === '% Committed' ? '' : '+');
        if(cur >= target) { el.textContent = target + (el.parentElement.querySelector('.stat-label')?.textContent === '% Committed' ? '' : '+'); clearInterval(t); }
      }, 25);
    });
    counterIO.unobserve(entry.target);
  }
}, { threshold: 0.5 });
counterIO.observe(document.getElementById('hero'));

// ============================================================
//  FAQ ACCORDION
// ============================================================
function toggleFaq(btn) {
  const item = btn.parentElement;
  const ans = item.querySelector('.faq-a');
  const isOpen = item.classList.contains('open');
  document.querySelectorAll('.faq-item').forEach(i => {
    i.classList.remove('open');
    i.querySelector('.faq-a').classList.remove('show');
  });
  if(!isOpen) { item.classList.add('open'); ans.classList.add('show'); }
}

// ============================================================
//  THEME TOGGLE
// ============================================================
function toggleTheme() {
  const html = document.documentElement;
  const isDark = html.getAttribute('data-theme') === 'dark';
  html.setAttribute('data-theme', isDark ? 'light' : 'dark');
  document.getElementById('theme-icon').className = isDark ? 'fas fa-moon' : 'fas fa-sun';
  saveData({ theme: isDark ? 'light' : 'dark' });
}
document.getElementById('theme-toggle').addEventListener('click', toggleTheme);

// ============================================================
function updateVisitor() {
  let count = parseInt(localStorage.getItem('portfolio_visitors') || '1000');
  count++;
  localStorage.setItem('portfolio_visitors', count);
  document.getElementById('visitor-count').textContent = count.toLocaleString();
}
